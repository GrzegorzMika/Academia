\begin{zd}

\end{zd}

\begin{zd}
Niech $W$ będzie procesem Wienera. Oblicz
\begin{itemize}
\item $\mathbb{E}(2W_2+3W_3+4W_4)$,
\item $\mathbb{P}(W_4-W_2>1|W_2=0.5)$.
\end{itemize}
\end{zd}

\end{zd}
